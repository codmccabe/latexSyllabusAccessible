\syllabussection{Course Learning Outcomes}

\begin{learningoutcomes}
  \item Define and identify integers, rational numbers, irrational numbers, and real numbers
  \item Apply set theory concepts to the real numbers
  \item Add, subtract, multiply, and divide signed numbers
  \item Determine powers, roots, and absolute values
  \item Identify and apply associative, commutative, and distributive properties
  \item Use the order of operations to evaluate expressions
  \item Evaluate algebraic expressions
  \item Express prime factorization of integers
  \item Apply the rules of exponents to algebraic and numerical expressions
  \item Convert between standard and scientific notation
  \item Express algebraic expressions in simplest form
  \item Solve linear equations with one variable
  \item Solve linear inequalities, graph solutions, and express in interval notation
  \item Solve literal equations and formulas
  \item Use linear equations and inequalities in problem solving
  \item Plot points and graph linear equations in two variables
  \item Determine the slope of a line
  \item Write equations of lines
  \item Solve linear systems in two or more variables
  \item Graph linear inequalities and systems of linear inequalities
  \item Add, subtract, multiply, divide, and simplify polynomials
  \item Factor polynomials using various techniques
  \item Solve equations using factoring
  \item Apply factoring to problem solving
\end{learningoutcomes}

\syllabussubsection{Grade Components}

\begin{gradecomponents}
  \item \textbf{Exams (50\% of grade)}
    \begin{taggeditemize}
      \item No drops and no retakes
    \end{taggeditemize}
  \item \textbf{Quizzes (20\% of grade)}
    \begin{taggeditemize}
      \item At least 3 lowest scores dropped
      \item No retakes or make-ups
    \end{taggeditemize}
  \item \textbf{ALEKS Homework (10\% of grade)}
    \begin{taggeditemize}
      \item At least 3 lowest scores dropped
      \item No extensions granted
    \end{taggeditemize}
  \item \textbf{Final Exam (20\% of grade)}
    \begin{taggeditemize}
      \item Cumulative exam covering entire semester
      \item Approximately 20 questions
      \item 2-hour time limit on scheduled exam day
    \end{taggeditemize}
\end{gradecomponents}
